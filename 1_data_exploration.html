def get_aged(age):
    if age < 13:
        return 0
    elif age < 20:
        return 1
    elif age < 30:
        return 2
    elif age < 45:
        return 3
    elif age < 60:
        return 4
    elif age >= 60:
        return 5
    else:
        return age
    
titanic['aged'] = titanic['Age'].apply(lambda x : get_aged(x))
